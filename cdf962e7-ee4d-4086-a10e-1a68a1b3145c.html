Integration Code

// Example: Fetch containers data
fetch('https://waste.qyzylorda.gov.kz/api/containers')
  .then(response => response.json())
  .then(data => console.log(data));

// Example: Report an issue
fetch('https://waste.qyzylorda.gov.kz/api/reports', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    containerId: 'TC-001',
    issue: 'Container is full',
    location: { lat: 44.8480, lon: 65.5000 }
  })
});
